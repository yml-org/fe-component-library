Link component With `props`

export const TemplateProps = (args) => html`<link-component
  .href=${args.href}
  .text=${args.text}
  .linkTitle=${args.linkTitle}
  .rel=${args.rel}
  .target=${args.target}
  .type=${args.type}
>
</link-component>`;
