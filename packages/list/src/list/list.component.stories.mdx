# List Component

The List component is a Lit component that renders a list component that is framework independent and utilizes Tailwind CSS internally.

# Examples

Ordered List default

export const orderedListTemplate = ({ listType, listItems }) =>
  html`<list-component .listType=${listType} .listItems=${listItems} />`;

<Canvas>
  <Story name="ordered list" args={orderedListVariant}>
    {orderedListTemplate.bind({})}
  </Story>
</Canvas>

Unordered List with default list styles overridden and bottom border

export const unOrderedListWithDefaultOverriddenAndBorderBottomTemplate = ({
  listType,
  listItems,
  showBorder,
  borderPosition,
  borderStyle,
  overrideDefaultListStyles,
}) =>
  html`<list-component
    .listType=${listType}
    .listItems=${listItems}
    .overrideDefaultListStyles=${overrideDefaultListStyles}
    .showBorder=${showBorder}
    .borderPosition=${borderPosition}
    .borderStyle=${borderStyle}
  />`;

<Canvas>
  <Story
    name="unordered list with default list styles overridden and border bottom"
    args={unOrderedListWithBottomBorder}
  >
    {unOrderedListWithDefaultOverriddenAndBorderBottomTemplate.bind({})}
  </Story>
</Canvas>
