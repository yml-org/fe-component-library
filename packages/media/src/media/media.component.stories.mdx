# Media Component

The Media component is a Lit component that renders a media component that is framework independent and utilizes Tailwind CSS internally.

# Props

| Prop Name             | Type   | Description                           |
| --------------------- | ------ | ------------------------------------- |
| mediaTitle            | string | to add title                          |
| imageSlotName         | string | the slot name for image slot          |
| staticContentSlotName | string | the slot name for static content slot |
| iconSlotName          | string | the slot name for icon slot           |
| mediaPartAttribute    | string | The part name for media component     |
| textPartAttribute     | string | The part name for text container      |
| titlePartAttribute    | string | The part name for title               |

# Examples

export const defaultMediaTemplate = ({ mediaTitle }) =>

Customized Media Component

export const customizedTemplate = ({
  mediaTitle,
  staticContentSlotName,
  imageSlotName,
  iconSlotName,
  mediaPartAttribute,
  textPartAttribute,
  titlePartAttribute,
}) =>
  html` <div class="parent-container">
    <media-component
      .mediaTitle=${mediaTitle}
      .imageSlotName=${imageSlotName}
      .staticContentSlotName=${staticContentSlotName}
      .iconSlotName=${iconSlotName}
      .mediaPartAttribute=${mediaPartAttribute}
      .textPartAttribute=${textPartAttribute}
      .titlePartAttribute=${titlePartAttribute}
    >
      <div slot=${imageSlotName}>${houseIcon}</div>
      <div class="static-slot-content" slot=${staticContentSlotName}>
        <p>Emotion</p>
        <p>Time</p>
      </div>
      <div class="icon-slot-content" slot=${iconSlotName}>${bookmarkIcon}</div>
    </media-component>
  </div>`;

<Canvas>
  <Story name="customized media component" args={mediaProps}>
    {customizedTemplate.bind({})}
  </Story>
</Canvas>
